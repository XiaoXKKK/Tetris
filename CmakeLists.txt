# CMake minimum version requirement
cmake_minimum_required(VERSION 3.10)

# Set C++ standard
set(CMAKE_CXX_STANDARD 20)

# Project name
project(Tetris)

# Add source files
set(SOURCES 
    main.cpp
    terminal.cpp
    terminal.h
    utils.cpp 
    utils.h
    draw.cpp
    draw.h
    define.h
    control.cpp
    control.h
    game.cpp
    game.h
    tetromino.cpp
    tetromino.h
    color.h
    piece.cpp
    piece.h
)

# Add executable target
add_executable(${PROJECT_NAME} ${SOURCES})
